def cleaned_visualise_word_map(x):
    words=" "
    for msg in x:
        msg = str(msg).lower()
        words = words+msg+" "
    wordcloud = WordCloud(stopwords = stopwords, width=3000, height=2500, background_color='white').generate(words)

    fig_size[1] = 7

    final = "".join(u for u in text if u not in ("?", ".", ";", ":",  "!",'"'))
